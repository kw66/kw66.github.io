<script>
  (function () {
    const reducedMotionQuery = window.matchMedia ? window.matchMedia('(prefers-reduced-motion: reduce)') : null;
    const prefersReducedMotion = reducedMotionQuery ? reducedMotionQuery.matches : false;
    const mascotBase = "{{ '/images/mascots/' | relative_url }}";
    const mascotSources = Array.from({ length: 45 }, (_, index) => {
      const name = `mascot-${String(index + 1).padStart(2, '0')}.png`;
      return `${mascotBase}${name}`;
    }).filter((src) => !src.endsWith('mascot-43.png'));
    const motionModeStorageKey = 'kw66-homepage-mascot-mode';
    const speedStorageKey = 'kw66-homepage-mascot-speed';
    const leftLayerId = 'side-mascot-layer-left';
    const rightLayerId = 'side-mascot-layer-right';
    const activePaperFilters = new Set(['first-author']);
    const featuredState = {
      index: 0,
      visibleCount: 1
    };
    const slotState = {
      spinning: false,
      timers: []
    };
    const mascotConfig = {
      minWidth: 1280,
      size: 84,
      gap: 84,
      speed: 34,
      minSpeedPercent: 0,
      maxSpeedPercent: 400,
      speedStepPercent: 25,
      leftInset: 0,
      rightInset: 0,
      preloadCount: 6,
      springAmplitude: 42,
      springIndexStep: 2.02,
      springTravelRate: 0.0104
    };
    const motionModes = {
      steady: {
        spring: false
      },
      spring: {
        spring: true
      }
    };
    let mascotRafId = null;
    let resizeTimer = null;
    let mascotState = null;
    let currentMotionMode = readStoredMotionMode();
    let currentSpeedPercent = readStoredSpeedPercent();

    function clamp(value, min, max) {
      return Math.min(Math.max(value, min), max);
    }

    function randomFrom(items) {
      return items[Math.floor(Math.random() * items.length)];
    }

    function readStoredMotionMode() {
      try {
        const stored = window.localStorage.getItem(motionModeStorageKey);
        return motionModes[stored] ? stored : 'steady';
      } catch (error) {
        return 'steady';
      }
    }

    function readStoredSpeedPercent() {
      try {
        const stored = Number(window.localStorage.getItem(speedStorageKey));
        if (Number.isFinite(stored)) {
          return clamp(stored, mascotConfig.minSpeedPercent, mascotConfig.maxSpeedPercent);
        }
      } catch (error) {
        // Ignore storage failures.
      }
      return prefersReducedMotion ? 0 : 100;
    }

    function storeMotionMode(mode) {
      try {
        window.localStorage.setItem(motionModeStorageKey, mode);
      } catch (error) {
        // Ignore storage failures.
      }
    }

    function storeSpeedPercent(speedPercent) {
      try {
        window.localStorage.setItem(speedStorageKey, String(speedPercent));
      } catch (error) {
        // Ignore storage failures.
      }
    }

    function resetPageScroll() {
      window.requestAnimationFrame(() => {
        window.scrollTo(0, 0);
        document.documentElement.scrollTop = 0;
        document.body.scrollTop = 0;
      });
    }

    function paintMascot(item) {
      item.el.style.transform = `translate3d(0, ${item.y.toFixed(2)}px, 0)`;
    }

    function respawnMascot(item) {
      item.el.src = randomFrom(mascotSources);
    }

    function springOffsetFor(lane, index) {
      const motion = motionModes[currentMotionMode] || motionModes.steady;
      if (!motion.spring) {
        return 0;
      }
      const phase = lane.phaseBias + index * mascotConfig.springIndexStep + lane.travel * mascotConfig.springTravelRate;
      return mascotConfig.springAmplitude * Math.sin(phase);
    }

    function buildLane(side, height, layer) {
      const baseSpacing = mascotConfig.size + mascotConfig.gap;
      const itemCount = Math.ceil(height / baseSpacing) + mascotConfig.preloadCount;
      const items = [];
      for (let index = 0; index < itemCount; index += 1) {
        const el = document.createElement('img');
        el.className = `side-mascot side-mascot--${side}`;
        el.alt = '';
        el.decoding = 'async';
        el.loading = 'eager';
        el.src = randomFrom(mascotSources);
        layer.appendChild(el);
        items.push({
          el,
          y: 0
        });
      }

      return {
        side,
        items,
        direction: side === 'left' ? 1 : -1,
        phaseBias: side === 'left' ? 0.38 : 1.26,
        baseSpacing,
        headY: -baseSpacing,
        sequenceStart: 0,
        travel: 0
      };
    }

    function paintLane(lane) {
      lane.items.forEach((item, index) => {
        const globalIndex = lane.sequenceStart + index;
        item.y = lane.headY + index * lane.baseSpacing + springOffsetFor(lane, globalIndex);
        paintMascot(item);
      });
    }

    function recycleLaneForward(lane) {
      const recycled = lane.items.pop();
      respawnMascot(recycled);
      lane.items.unshift(recycled);
      lane.sequenceStart -= 1;
      lane.headY -= lane.baseSpacing;
    }

    function recycleLaneBackward(lane) {
      const recycled = lane.items.shift();
      respawnMascot(recycled);
      lane.items.push(recycled);
      lane.sequenceStart += 1;
      lane.headY += lane.baseSpacing;
    }

    function advanceLane(lane, distance) {
      lane.travel += distance;
      lane.headY += lane.direction * distance;

      if (lane.direction === 1) {
        while (lane.headY >= 0) {
          recycleLaneForward(lane);
        }
        return;
      }

      while (lane.headY <= -lane.baseSpacing) {
        recycleLaneBackward(lane);
      }
    }

    function mascotStep(now) {
      if (!mascotState) return;
      const dt = Math.min((now - mascotState.lastTime) / 1000, 0.05);
      mascotState.lastTime = now;
      const distance = mascotState.speed * (currentSpeedPercent / 100) * dt;

      advanceLane(mascotState.left, distance);
      advanceLane(mascotState.right, distance);
      paintLane(mascotState.left);
      paintLane(mascotState.right);

      mascotRafId = window.requestAnimationFrame(mascotStep);
    }

    function renderSideMascots() {
      const leftLayer = document.getElementById(leftLayerId);
      const rightLayer = document.getElementById(rightLayerId);
      if (!leftLayer || !rightLayer) return;

      leftLayer.innerHTML = '';
      rightLayer.innerHTML = '';
      if (mascotRafId) {
        window.cancelAnimationFrame(mascotRafId);
        mascotRafId = null;
      }
      mascotState = null;

      if (window.innerWidth < mascotConfig.minWidth) return;

      const railHeight = Math.max(
        document.documentElement.scrollHeight,
        document.body.scrollHeight,
        window.innerHeight,
        1080
      );

      mascotState = {
        speed: mascotConfig.speed,
        lastTime: performance.now(),
        left: buildLane('left', railHeight, leftLayer),
        right: buildLane('right', railHeight, rightLayer)
      };

      mascotState.left.items.forEach((item) => {
        item.el.style.width = `${mascotConfig.size}px`;
        item.el.style.left = `${mascotConfig.leftInset}px`;
      });
      mascotState.right.items.forEach((item) => {
        item.el.style.width = `${mascotConfig.size}px`;
        item.el.style.right = `${mascotConfig.rightInset}px`;
      });

      paintLane(mascotState.left);
      paintLane(mascotState.right);
      mascotRafId = window.requestAnimationFrame(mascotStep);
    }

    function updateMotionButtons() {
      document.querySelectorAll('[data-mascot-mode]').forEach((button) => {
        const isActive = button.dataset.mascotMode === currentMotionMode;
        button.classList.toggle('is-active', isActive);
        button.setAttribute('aria-pressed', String(isActive));
      });
    }

    function updateSpeedControls() {
      const readout = document.getElementById('mascot-speed-readout');
      if (readout) {
        readout.textContent = `${currentSpeedPercent}%`;
      }
      document.querySelectorAll('[data-speed-action]').forEach((button) => {
        const isDecrease = button.dataset.speedAction === 'decrease';
        const disabled = isDecrease
          ? currentSpeedPercent <= mascotConfig.minSpeedPercent
          : currentSpeedPercent >= mascotConfig.maxSpeedPercent;
        button.disabled = disabled;
        button.classList.toggle('is-disabled', disabled);
      });
    }

    function setMotionMode(mode) {
      if (!motionModes[mode]) return;
      currentMotionMode = mode;
      storeMotionMode(mode);
      updateMotionButtons();
      if (mascotState) {
        paintLane(mascotState.left);
        paintLane(mascotState.right);
      }
    }

    function setSpeedPercent(nextPercent) {
      currentSpeedPercent = clamp(nextPercent, mascotConfig.minSpeedPercent, mascotConfig.maxSpeedPercent);
      storeSpeedPercent(currentSpeedPercent);
      updateSpeedControls();
    }

    function initMotionControls() {
      updateMotionButtons();
      updateSpeedControls();
      document.querySelectorAll('[data-mascot-mode]').forEach((button) => {
        button.addEventListener('click', () => {
          setMotionMode(button.dataset.mascotMode);
        });
      });
      document.querySelectorAll('[data-speed-action]').forEach((button) => {
        button.addEventListener('click', () => {
          const delta = button.dataset.speedAction === 'increase'
            ? mascotConfig.speedStepPercent
            : -mascotConfig.speedStepPercent;
          setSpeedPercent(currentSpeedPercent + delta);
        });
      });
    }

    function updatePaperFilterButtons() {
      const allActive = activePaperFilters.size === 0;
      document.querySelectorAll('[data-filter-tag]').forEach((button) => {
        const tag = button.dataset.filterTag;
        const isActive = tag === 'all' ? allActive : activePaperFilters.has(tag);
        button.classList.toggle('is-active', isActive);
        button.setAttribute('aria-pressed', String(isActive));
      });
    }

    function applyPaperFilters() {
      const items = Array.from(document.querySelectorAll('.paper-list-item[data-paper-tags]'));
      const emptyState = document.getElementById('paper-filter-empty');
      const activeTags = Array.from(activePaperFilters);
      let visibleCount = 0;

      items.forEach((item) => {
        const tags = item.dataset.paperTags.split(',').map((tag) => tag.trim()).filter(Boolean);
        const matched = activeTags.length === 0 || activeTags.every((tag) => tags.includes(tag));
        item.hidden = !matched;
        if (matched) {
          visibleCount += 1;
        }
      });

      if (emptyState) {
        emptyState.hidden = visibleCount !== 0;
      }
      updatePaperFilterButtons();
    }

    function initPaperFilters() {
      document.querySelectorAll('[data-filter-tag]').forEach((button) => {
        button.addEventListener('click', () => {
          const tag = button.dataset.filterTag;
          if (tag === 'all') {
            activePaperFilters.clear();
          } else if (activePaperFilters.has(tag)) {
            activePaperFilters.delete(tag);
          } else {
            activePaperFilters.add(tag);
          }
          applyPaperFilters();
        });
      });
      applyPaperFilters();
    }

    function getFeaturedVisibleCount() {
      if (window.innerWidth >= 1280) return 3;
      if (window.innerWidth >= 860) return 2;
      return 1;
    }

    function updateFeaturedButtons() {
      const maxIndex = Math.max(0, document.querySelectorAll('[data-featured-item]').length - featuredState.visibleCount);
      document.querySelectorAll('[data-featured-nav]').forEach((button) => {
        const action = button.dataset.featuredNav;
        const disabled = action === 'prev' ? featuredState.index <= 0 : featuredState.index >= maxIndex;
        button.disabled = disabled;
        button.classList.toggle('is-disabled', disabled);
      });
    }

    function scrollFeatured(behavior) {
      const viewport = document.getElementById('featured-viewport');
      const items = Array.from(document.querySelectorAll('[data-featured-item]'));
      if (!viewport || !items.length) return;
      const target = items[featuredState.index];
      if (!target) return;
      viewport.scrollTo({
        left: target.offsetLeft,
        top: 0,
        behavior
      });
      updateFeaturedButtons();
    }

    function syncFeaturedCarousel(behavior) {
      const track = document.getElementById('featured-track');
      const items = Array.from(document.querySelectorAll('[data-featured-item]'));
      if (!track || !items.length) return;
      featuredState.visibleCount = Math.min(getFeaturedVisibleCount(), items.length);
      featuredState.index = clamp(
        featuredState.index,
        0,
        Math.max(0, items.length - featuredState.visibleCount)
      );
      track.style.setProperty('--featured-visible', String(featuredState.visibleCount));
      scrollFeatured(behavior);
    }

    function initFeaturedCarousel() {
      syncFeaturedCarousel('auto');
      document.querySelectorAll('[data-featured-nav]').forEach((button) => {
        button.addEventListener('click', () => {
          const maxIndex = Math.max(0, document.querySelectorAll('[data-featured-item]').length - featuredState.visibleCount);
          if (button.dataset.featuredNav === 'prev') {
            featuredState.index = clamp(featuredState.index - 1, 0, maxIndex);
          } else {
            featuredState.index = clamp(featuredState.index + 1, 0, maxIndex);
          }
          scrollFeatured(prefersReducedMotion ? 'auto' : 'smooth');
        });
      });
    }

    function createSlotItem(src) {
      const item = document.createElement('div');
      item.className = 'slot-machine-item';
      const img = document.createElement('img');
      img.src = src;
      img.alt = '';
      img.decoding = 'async';
      img.loading = 'eager';
      item.appendChild(img);
      return item;
    }

    function renderSlotTrack(track, sources) {
      track.innerHTML = '';
      sources.forEach((src) => {
        track.appendChild(createSlotItem(src));
      });
    }

    function setSlotResult(track, src) {
      renderSlotTrack(track, [src]);
      track.style.transition = 'none';
      track.style.transform = 'translate3d(0, 0, 0)';
      track.dataset.currentSource = src;
    }

    function clearSlotTimers() {
      slotState.timers.forEach((timerId) => {
        window.clearTimeout(timerId);
      });
      slotState.timers = [];
    }

    function spinSlotMachine() {
      const slotMachine = document.getElementById('mascot-slot-machine');
      const lever = document.getElementById('slot-machine-lever');
      const reels = Array.from(document.querySelectorAll('[data-slot-reel]'));
      const tracks = Array.from(document.querySelectorAll('[data-slot-track]'));
      if (!slotMachine || !lever || !reels.length || slotState.spinning) return;

      clearSlotTimers();
      slotState.spinning = true;
      slotMachine.classList.add('is-spinning');
      lever.classList.add('is-pulled');
      let settledCount = 0;

      tracks.forEach((track, index) => {
        const reel = reels[index];
        const sequenceLength = prefersReducedMotion ? 6 + index : 16 + index * 5;
        const sequence = [];
        for (let pointer = 0; pointer < sequenceLength; pointer += 1) {
          sequence.push(randomFrom(mascotSources));
        }
        const target = randomFrom(mascotSources);
        sequence.push(target);

        renderSlotTrack(track, sequence);
        track.style.transition = 'none';
        track.style.transform = 'translate3d(0, 0, 0)';
        track.offsetWidth;

        const duration = prefersReducedMotion ? 80 : 760 + index * 320;
        const stepWidth = reel.clientWidth;
        window.requestAnimationFrame(() => {
          track.style.transition = `transform ${duration}ms cubic-bezier(0.08, 0.86, 0.18, 1)`;
          track.style.transform = `translate3d(${-(sequence.length - 1) * stepWidth}px, 0, 0)`;
        });

        let settled = false;
        const handleSettle = () => {
          if (settled) return;
          settled = true;
          track.removeEventListener('transitionend', handleSettle);
          setSlotResult(track, target);
          settledCount += 1;
          if (settledCount === tracks.length) {
            slotState.spinning = false;
            slotMachine.classList.remove('is-spinning');
            const releaseTimer = window.setTimeout(() => {
              lever.classList.remove('is-pulled');
            }, prefersReducedMotion ? 30 : 200);
            slotState.timers.push(releaseTimer);
          }
        };

        track.addEventListener('transitionend', handleSettle);
        const fallbackTimer = window.setTimeout(handleSettle, duration + 160);
        slotState.timers.push(fallbackTimer);
      });
    }

    function initSlotMachine() {
      const lever = document.getElementById('slot-machine-lever');
      const tracks = Array.from(document.querySelectorAll('[data-slot-track]'));
      if (!lever || !tracks.length) return;
      tracks.forEach((track) => {
        setSlotResult(track, randomFrom(mascotSources));
      });
      lever.addEventListener('click', spinSlotMachine);
    }

    function handleResize() {
      window.clearTimeout(resizeTimer);
      resizeTimer = window.setTimeout(() => {
        renderSideMascots();
        syncFeaturedCarousel('auto');
      }, 160);
    }

    if ('scrollRestoration' in window.history) {
      window.history.scrollRestoration = 'manual';
    }
    resetPageScroll();
    window.addEventListener('pageshow', resetPageScroll);

    initMotionControls();
    initPaperFilters();
    initFeaturedCarousel();
    initSlotMachine();

    if (document.readyState === 'complete') {
      renderSideMascots();
      syncFeaturedCarousel('auto');
    } else {
      window.addEventListener('load', () => {
        resetPageScroll();
        renderSideMascots();
        syncFeaturedCarousel('auto');
      }, { once: true });
    }

    window.addEventListener('resize', handleResize);
    window.addEventListener('beforeunload', () => {
      if (mascotRafId) {
        window.cancelAnimationFrame(mascotRafId);
      }
      clearSlotTimers();
    });
  })();
</script>
